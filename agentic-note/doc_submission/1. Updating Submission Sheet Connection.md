        // Helper to parse CSV string (handling quotes, commas, and linebreaks)
        function parseCSV(str) {
            const arr = [];
            let quote = false;
            let col = 0, row = 0;

            for (let c = 0; c < str.length; c++) {
                let cc = str[c], nc = str[c + 1];
                arr[row] = arr[row] || [];
                arr[row][col] = arr[row][col] || '';

                if (cc == '"' && quote && nc == '"') {
                    arr[row][col] += cc; ++c; continue;
                }
                if (cc == '"') {
                    quote = !quote; continue;
                }
                if (cc == ',' && !quote) {
                    ++col; continue;
                }
                if (cc == '\r' && nc == '\n' && !quote) {
                    ++row; col = 0; ++c; continue;
                }
                if (cc == '\n' && !quote) {
                    ++row; col = 0; continue;
                }
                if (cc == '\r' && !quote) {
                    ++row; col = 0; continue;
                }
                arr[row][col] += cc;
            }
            return arr;
        }
